\section{Hình vẽ TikZ}

\begin{center}
\begin{tikzpicture}[scale=1.5]
    % Axes
    \draw[->] (-0.5,0) -- (4,0) node[right] {$x$};
    \draw[->] (0,-0.5) -- (0,3) node[above] {$y$};
    
    % Function curve
    \draw[thick,blue,domain=0:3.5] plot (\x,{0.5*\x*\x - \x + 1});
    
    % Points
    \fill[red] (1,0.5) circle (2pt) node[below right] {$A$};
    \fill[red] (2,1) circle (2pt) node[above right] {$B$};
    
    % Grid
    \draw[gray,very thin] (0,0) grid (3.5,2.5);
\end{tikzpicture}
\end{center}
