\documentclass[border=2mm]{standalone}
\usepackage{tikz}
\usetikzlibrary{calc,patterns,angles,quotes}
\begin{document}
\begin{tikzpicture}[scale=0.8]
    % Axes
    \draw[->] (-7,0) -- (5,0) node[right] {$x$};
    \draw[->] (0,-6) -- (0,6) node[above] {$y$};
    \node at (0,0) [below left] {$O$};
    % Asymptotes
    \draw[dashed, blue, thick] (-2,-4) -- (-2,4) node[above] {$x=-2$};
    \draw[dashed, blue, thick] (-5,1) -- (5,1) node[right] {$y=1$};
    % Plot
    \draw[domain=-6:-2.5
, smooth, variable=\x, red, thick] plot ({\x}, {(\x-2)/(\x+2)});
    \draw[domain=-1.55:5, smooth, variable=\x, red, thick] plot ({\x}, {(\x-2)/(\x+2)});
    % Points
    \fill[blue] (0,-1) circle (3pt) node[above right] {$(0,-1)$};
    \fill[blue] (2,0) circle (3pt) node[below right] {$(2,0)$};
    \fill[blue] (-2,1) circle (3pt) node[above left] {$(-2,1)$};
    % Ticks
    \foreach \x in {-4,-2,2,4}
        \draw (\x,0.1) -- (\x,-0.1) node[below] {$\x$};
    \foreach \y in {-3,-1,1,3}
        \draw (0.1,\y) -- (-0.1,\y) node[left] {$\y$};
\end{tikzpicture}
\end{document}
